// src/utils/api.ts
const API_BASE_URL = import.meta.env.VITE_API_BASE_URL || '/api'

interface ApiResponse<T> {
  success: boolean
  message?: string
  data?: T
}

async function apiRequest<T>(
  endpoint: string,
  options: RequestInit = {}
): Promise<ApiResponse<T>> {
  const url = `${API_BASE_URL}${endpoint}`
  
  const response = await fetch(url, {
    ...options,
    headers: {
      'Content-Type': 'application/json',
      ...options.headers,
    },
  })
  
  if (!response.ok) {
    throw new Error(`API Error: ${response.statusText}`)
  }
  
  return response.json()
}

export const api = {
  // Contact
  submitContact: (data: ContactFormData) =>
    apiRequest('/contact/submit', {
      method: 'POST',
      body: JSON.stringify(data),
    }),
  
  // Newsletter
  subscribeNewsletter: (email: string) =>
    apiRequest('/newsletter/subscribe', {
      method: 'POST',
      body: JSON.stringify({ email }),
    }),
  
  // Product Verification
  verifyProduct: (code: string) =>
    apiRequest('/product/verify', {
      method: 'POST',
      body: JSON.stringify({ code }),
    }),
  
  // Insights
  getInsights: (page = 1, limit = 10, category = 'all') =>
    apiRequest(`/insights?page=${page}&limit=${limit}&category=${category}`),
  
  getInsightBySlug: (slug: string) =>
    apiRequest(`/insights/${slug}`),
  
  // Careers
  getPositions: (lang = 'en') =>
    apiRequest(`/careers/positions?lang=${lang}`),
  
  submitApplication: (data: ApplicationFormData) =>
    apiRequest('/careers/apply', {
      method: 'POST',
